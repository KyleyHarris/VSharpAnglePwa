Integration Example

<!-- HTML -->
<knife-notation-input 
  notation="in=>15dps-0.003w,10dps-0.015w"
  placeholder="Enter knife notation..."
  @model-parsed="${handleModelParsed}"
  @notation-submit="${handleSubmit}">
</knife-notation-input>

  const { notation, model } = e.detail;
  console.log('Notation submitted:', notation);

}
